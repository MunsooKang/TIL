1. npm install tailwindcss postcss-cli autoprefixer
2. npx tailwindcss init -p
3. ``` // tailwind.config.js
/** @type {import('tailwindcss').Config} */

module.exports = {
    content: ["./src/**/*.{js,jsx,ts,tsx}"],
    theme: {
        extend: {},
    },
    plugins: [],
};
```
4. create src => tailwind.css
```
@tailwind base;
@tailwind components;
@tailwind utilities;
```

5. // App.jsx
import "./tailwind.css"
